# WebAR via QR-Code Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** QR-Code-Icon neben der Möbel-ID. Klick → Modal mit QR. Handy scannt → AR-Seite baut 3D live → model-viewer mit AR-Platzierung.

**Architecture:** QR-Code Modal als React Component. AR-Seite als standalone Next.js Page. Three.js GLTFExporter für GLB-Export client-seitig. Google model-viewer als CDN Script.

**Tech Stack:** Next.js 16, Three.js GLTFExporter, qrcode (npm), Google model-viewer (CDN)

```tsx
'use client';

import { useEffect, useRef } from 'react';
import QRCode from 'qrcode';

interface QRCodeModalProps {
  moebelId: number;
  onClose: () => void;
}

export default function QRCodeModal({ moebelId, onClose }: QRCodeModalProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const arUrl = `${window.location.origin}/ar?config=${moebelId}`;

  useEffect(() => {
    if (canvasRef.current) {
      void QRCode.toCanvas(canvasRef.current, arUrl, {
        width: 256,
        margin: 2,
        color: { dark: '#1C1A17', light: '#FFFFFF' },
      });
    }
  }, [arUrl]);

  return (
    <div
      style={{
        position: 'fixed', inset: 0, zIndex: 200,
        background: 'rgba(0,0,0,0.55)',
        display: 'flex', alignItems: 'center', justifyContent: 'center',
      }}
      onClick={e => { if (e.target === e.currentTarget) onClose(); }}
    >
      <div
        style={{
          background: '#fff', borderRadius: 16, padding: '32px 36px',
          maxWidth: 340, width: '100%', textAlign: 'center',
          boxShadow: '0 8px 32px rgba(0,0,0,0.18)',
        }}
      >
        {/* Schließen */}
        <button
          onClick={onClose}
          style={{
            position: 'absolute', top: 12, right: 16,
            background: 'none', border: 'none', fontSize: 20,
            color: '#999', cursor: 'pointer',
          }}
        >×</button>

        <div style={{ fontSize: 14, fontWeight: 600, color: '#1C1A17', marginBottom: 4 }}>
          AR-Vorschau
        </div>
        <div style={{ fontSize: 11, color: '#7A7670', marginBottom: 16 }}>
          Möbel #{String(moebelId).padStart(8, '0')}
        </div>

        {/* QR-Code */}
        <canvas ref={canvasRef} style={{ margin: '0 auto', display: 'block', borderRadius: 8 }} />

        <div style={{ fontSize: 11, color: '#7A7670', marginTop: 16, lineHeight: 1.6 }}>
          QR-Code mit dem Handy scannen.<br />
          Das Möbel wird mit korrekten Maßen<br />
          in Ihrem Raum angezeigt.
        </div>

        {/* URL zum Kopieren */}
        <button
          onClick={() => { void navigator.clipboard.writeText(arUrl); }}
          style={{
            marginTop: 12, padding: '6px 16px', borderRadius: 6,
            border: '1px solid #E8E5E0', background: '#FAFAF8',
            fontSize: 10, color: '#7A7670', cursor: 'pointer',
          }}
        >
          Link kopieren
        </button>
      </div>
    </div>
  );
}
```

- [ ] **Step 2: AR Client Component (ARViewer.tsx)**

```tsx
'use client';

import { useEffect, useState, useRef, useCallback } from 'react';
import { useSearchParams } from 'next/navigation';
import * as THREE from 'three';
import { GLTFExporter } from 'three/examples/jsm/exporters/GLTFExporter.js';
import type { ConfigState } from '@/core/types';
import { useModuleGeometry } from '@/features/preview3d/useModuleGeometry';
import { MAT_BY_V, MATERIALS } from '@/core/constants';
import Script from 'next/script';

type ViewerState = 'loading' | 'building' | 'ready' | 'error';

export default function ARViewer() {
  const searchParams = useSearchParams();
  const configCode = searchParams.get('config');

  const [config, setConfig] = useState<ConfigState | null>(null);
  const [viewerState, setViewerState] = useState<ViewerState>('loading');
  const [glbUrl, setGlbUrl] = useState<string | null>(null);
  const [errorMsg, setErrorMsg] = useState('');
  const modelViewerRef = useRef<HTMLElement>(null);

  // Konfiguration aus Supabase laden
  useEffect(() => {
    if (!configCode) {
      setViewerState('error');
      setErrorMsg('Keine Möbel-ID angegeben');
      return;
    }
    fetch(`/api/config/load?code=${configCode}`)
      .then(res => {
        if (!res.ok) throw new Error('not found');
        return res.json();
      })
      .then((data: { config?: ConfigState }) => {
        if (!data.config) throw new Error('invalid');
        setConfig(data.config);
        setViewerState('building');
      })
      .catch(() => {
        setViewerState('error');
        setErrorMsg('Konfiguration nicht gefunden');
      });
  }, [configCode]);

  // Szene-Objekte aus Config berechnen
  const sceneObjects = useModuleGeometry(config ?? {
    cols: [580], rows: [360], depth: 360, grid: [[{ type: '', shelves: 0 }]],
    surface: 'none', handle: 'none', footer: 'none',
    opts: { outer: false, inner: false, back: false },
    bomOverrides: {}, cableHoles: {}, catOverrides: {},
  });

  // GLB exportieren wenn Config geladen
  const buildGLB = useCallback(async () => {
    if (!config || sceneObjects.length === 0) return;

    const scene = new THREE.Scene();

    for (const obj of sceneObjects) {
      const geo = new THREE.BoxGeometry(...obj.size);

      // Farbe bestimmen
      let color = obj.color;
      const mat = MATERIALS.find(m => m.hex === color);

      const material = new THREE.MeshStandardMaterial({
        color: new THREE.Color(color).convertSRGBToLinear(),
        roughness: obj.partType === 'profil' ? 0.18 : 0.72,
        metalness: obj.partType === 'profil' ? 0.85 : 0.0,
      });

      const mesh = new THREE.Mesh(geo, material);
      mesh.position.set(...obj.position);
      if (obj.rotation) mesh.rotation.set(...obj.rotation);
      mesh.castShadow = true;
      mesh.receiveShadow = true;
      scene.add(mesh);
    }

    // Skalierung: Three.js-Einheiten (0.01 = 1mm) → Meter (÷ 100)
    scene.scale.set(0.1, 0.1, 0.1);

    const exporter = new GLTFExporter();

    return new Promise<string>((resolve, reject) => {
      exporter.parse(
        scene,
        (result) => {
          const blob = new Blob(
            [result instanceof ArrayBuffer ? result : JSON.stringify(result)],
            { type: 'model/gltf-binary' }
          );
          resolve(URL.createObjectURL(blob));
        },
        reject,
        { binary: true }
      );
    });
  }, [config, sceneObjects]);

  // GLB bauen wenn Szene bereit
  useEffect(() => {
    if (viewerState !== 'building' || !config) return;

    buildGLB()
      .then(url => {
        if (url) {
          setGlbUrl(url);
          setViewerState('ready');
        }
      })
      .catch(() => {
        setViewerState('error');
        setErrorMsg('3D-Modell konnte nicht erstellt werden');
      });
  }, [viewerState, config, buildGLB]);

  // Fehler-Anzeige
  if (viewerState === 'error') {
    return (
      <div style={{
        width: '100vw', height: '100vh', display: 'flex',
        alignItems: 'center', justifyContent: 'center',
        background: '#F5F2ED', fontFamily: '-apple-system, Arial, sans-serif',
      }}>
        <div style={{ textAlign: 'center', padding: 24 }}>
          <div style={{ fontSize: 40, marginBottom: 16 }}>⚠</div>
          <div style={{ fontSize: 14, color: '#1C1A17', fontWeight: 600, marginBottom: 8 }}>
            {errorMsg}
          </div>
          <a href="/" style={{ fontSize: 12, color: '#8A7050' }}>
            Zurück zum Konfigurator
          </a>
        </div>
      </div>
    );
  }

  // Lade-Anzeige
  if (viewerState !== 'ready' || !glbUrl) {
    return (
      <div style={{
        width: '100vw', height: '100vh', display: 'flex',
        alignItems: 'center', justifyContent: 'center',
        background: '#F5F2ED', fontFamily: '-apple-system, Arial, sans-serif',
      }}>
        <div style={{ textAlign: 'center' }}>
          <div style={{ fontSize: 16, fontWeight: 600, color: '#1C1A17', marginBottom: 8 }}>
            Artmodul AR
          </div>
          <div style={{ fontSize: 12, color: '#7A7670' }}>
            {viewerState === 'loading' ? 'Lade Konfiguration…' : 'Erstelle 3D-Modell…'}
          </div>
        </div>
      </div>
    );
  }

  return (
    <>
      <Script
        src="https://ajax.googleapis.com/ajax/libs/model-viewer/3.5.0/model-viewer.min.js"
        type="module"
      />
      <div style={{ width: '100vw', height: '100vh', background: '#F5F2ED' }}>
        {/* @ts-expect-error — model-viewer ist ein Web Component, kein React-Element */}
        <model-viewer
          ref={modelViewerRef}
          src={glbUrl}
          ar
          ar-modes="webxr scene-viewer quick-look"
          camera-controls
          touch-action="pan-y"
          auto-rotate
          shadow-intensity="1"
          environment-image="neutral"
          style={{ width: '100%', height: '100%' }}
        >
          <button
            slot="ar-button"
            style={{
              position: 'absolute', bottom: 24, left: '50%', transform: 'translateX(-50%)',
              padding: '12px 28px', borderRadius: 12, border: 'none',
              background: '#1C1A17', color: '#fff', fontSize: 14, fontWeight: 600,
              cursor: 'pointer', boxShadow: '0 4px 16px rgba(0,0,0,0.2)',
            }}
          >
            In meinem Raum ansehen
          </button>
          <div
            slot="poster"
            style={{
              width: '100%', height: '100%', display: 'flex',
              alignItems: 'center', justifyContent: 'center',
              fontSize: 12, color: '#7A7670',
            }}
          >
            Lade 3D-Modell…
          </div>
        </model-viewer>

        {/* Header-Overlay */}
        <div style={{
          position: 'absolute', top: 0, left: 0, right: 0, padding: '16px 20px',
          display: 'flex', justifyContent: 'space-between', alignItems: 'center',
          background: 'linear-gradient(rgba(0,0,0,0.3), transparent)',
        }}>
          <div>
            <div style={{ fontSize: 14, fontWeight: 600, color: '#fff' }}>Artmodul</div>
            <div style={{ fontSize: 10, color: 'rgba(255,255,255,0.7)' }}>
              Möbel #{configCode}
            </div>
          </div>
          <a
            href="/"
            style={{
              padding: '6px 14px', borderRadius: 8,
              background: 'rgba(255,255,255,0.15)',
              color: '#fff', fontSize: 11, textDecoration: 'none',
              backdropFilter: 'blur(8px)',
            }}
          >
            Konfigurator
          </a>
        </div>
      </div>
    </>
  );
}
```
